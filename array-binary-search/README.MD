# Binary search in a sorted 1D array

## Solution

def binary_search(arr, key):

    begin = 0
    end = len(arr) - 1
    
    while begin <= end:
        mid = (begin + end) // 2
        
        if arr[mid] == key:
            return mid
        elif arr[mid] < key:
            begin = mid + 1
        else:
            end = mid - 1
    
    return -1
